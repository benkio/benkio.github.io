  $("#totalArticleCount").text("Total Articles: " + htmlArticlesPaths.length);

  const siteOrigin = window.location.origin;

  htmlArticlesPaths.forEach((value, index) => {
    let pageUrl = siteOrigin + value.slice(1);
    fetch(pageUrl)
      .then((response) => {
        return response.text();
      })
      .then((html) => {
        const parser = new DOMParser();
        const doc = parser.parseFromString(html, "text/html");
        const articleTitle = doc.querySelector("#Article").innerHTML;
        $("#ArticleList").append(
          `<a href="${pageUrl}" class="w-100 w-md-75 m-auto list-group-item list-group-item-action article">${articleTitle}</a>`,
        );
      });
  });

  /* Attributes and classes that I can't add by plain org */

  document.getElementById("content").classList.add("container-fluid", "p-0");
  document.getElementById("text-navbar").classList.add("container-fluid");
  document
    .getElementById("outline-container-navbar")
    .setAttribute("data-bs-theme", "dark");
#+end_src
